grafico1 <- 
pokemon |>
ggplot(aes(x = Attack, y = Defense,
           color = `generacion`,
            text = paste0(
      "Nombre: ", Name, "\n")
      )
      ) +
  geom_point(size = 1.3) +
    scale_x_continuous(labels = label_comma(), limits = c(0, 200)) +
  ggtitle("Relacion entre puntos de ataque y defensa de los pokemon") +
  xlab("Ataque") +
  ylab("Defensa") +
  labs(caption = "Fuentes: Pokemon Data",
  color = "Generacion") +
  scale_colour_brewer(palette = "Set2") +
  theme_minimal()

ggplotly(grafico1)

agua <-
pokemon |>
  filter(`Type 1` == "Water") |>
  group_by(Generation, Total) |>
  count(Generation)
Código 
grafico3 <-
  agua |>
  ggplot(aes(x = reorder(Generation, Total, median),
                y = Total)) +
  geom_boxplot(color = "#123456", fill = "#8AE550") +
  ggtitle("Total de puntos base para pokemon tipo agua por generación") +
  xlab("Generación") +
  ylab("Total")

ggplotly(grafico3) |>
  config(locale = 'es')
